LOCALHOST:

wp_blogs -> domain: localhost | path: /website/
         -> domain: localhost | path: /website/st-blog/

wp_options -> siteurl: http://localhost/website
           -> home: http://localhost/website

wp_site -> domain: localhost | path: /website/

wp_2_options -> siteurl: http://localhost/website/st-blog
             -> home: http://localhost/website

           -> home: http://servitalleres.com/
